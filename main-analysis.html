    for (i in 1:nrow(stats_df)) {
      row_name <- rownames(stats_df)[i]
      row <- stats_df[i, ]
      
      df_val <- row[["Df"]]
      pillai <- row[["Pillai"]]
      approx_f <- row[["approx F"]]
      num_df <- row[["num Df"]]
      den_df <- row[["den Df"]]
      p_val <- row[["Pr(>F)"]]
      
      pillai_str <- if (is.na(pillai)) "" else sprintf("%.5f", pillai)
      f_str <- if (is.na(approx_f)) "" else sprintf("%.3f", approx_f)
      num_df_str <- if (is.na(num_df)) "" else as.character(num_df)
      den_df_str <- if (is.na(den_df)) "" else as.character(den_df)
      
      p_str <- ""
      sig_class <- ""
      stars <- ""
      if (!is.na(p_val)) {
        p_str <- if (p_val < 0.0001) "&lt; 0.0001" else sprintf("%.5f", p_val)
        if (p_val < 0.001) {
          stars <- "***"
          sig_class <- "class='sig'"
        } else if (p_val < 0.01) {
          stars <- "**"
          sig_class <- "class='sig'"
        } else if (p_val < 0.05) {
          stars <- "*"
          sig_class <- "class='sig'"
        } else if (p_val < 0.1) {
          stars <- "."
          sig_class <- "class='ns'"
        } else {
          stars <- "n.s."
          sig_class <- "class='ns'"
        }
      }

        html_content,
        "    <tr>\n",

    # Post-hoc pairwise comparisons for race/ethnicity
    levels_race <- levels(mfa.dat$race.f)
    pairs <- combn(levels_race, 2)
    results <- list()
    for (i in 1:ncol(pairs)) {
      g1 <- pairs[1, i]
      g2 <- pairs[2, i]
      sub_dat <- mfa.dat |> dplyr::filter(race.f %in% c(g1, g2))
      fit <- manova(cbind(Dim.1, Dim.2, Dim.3) ~ race.f, data = sub_dat)
      results[[i]] <- data.frame(
        Comparison = paste(g1, "vs.", g2),
        Pillai = summary(fit, test = "Pillai")$stats[1, "Pillai"],
        p_value = summary(fit, test = "Pillai")$stats[1, "Pr(>F)"]
      )
    }
    pairwise_df <- do.call(rbind, results) |> 
      mutate(p_adj_bonferroni = p.adjust(p_value, method = "bonferroni")) |> 
      arrange(p_value)
    
    writeLines(
        knitr::kable(pairwise_df, format = "html", caption = "Table 2. Post-hoc Pairwise MANOVA Comparisons for Race/Ethnicity Groups"), 
        here("Tabs", "pairwise_manovas.html")
    )
    
    # Univariate Tukey HSD on Dimension 1
    tukey_d1 <- TukeyHSD(aov(Dim.1 ~ race.f, data = mfa.dat))
    writeLines(
        knitr::kable(as.data.frame(tukey_d1$race.f), format = "html", caption = "Table 3. Tukey's Honestly Significant Difference (HSD) pairwise results on Dimension 1 (Fancy Taste) for Race/Ethnicity"), 
        here("Tabs", "tukey_hsd_dim1.html")
    )

    # Run HCPC clustering (using 3 clusters)
    hcpc_res <- HCPC(mfa.res, nb.clust = 3, graph = FALSE)
    mfa.dat$Cluster <- hcpc_res$data.clust$clust
    
    # Intuitive labels for clusters
    cluster_labels <- c(
        "1" = "Everyday Convenience Seekers",
        "2" = "Traditional/Conservative Homebody",
        "3" = "Culinary Explorer/High-End Enthusiast"
    )
    mfa.dat$ClusterName <- factor(cluster_labels[as.character(mfa.dat$Cluster)], levels = cluster_labels)
    
    # Calculate centroids for plotting
    centroids_plot <- mfa.dat |> 
        group_by(ClusterName) |> 
        summarize(Dim.1 = mean(Dim.1), Dim.2 = mean(Dim.2))

    # Custom cloud plot using ggplot2 following aesthetic_politics style
    p_hcpc_map <- ggplot(mfa.dat, aes(x = Dim.1, y = Dim.2, color = ClusterName)) +
        geom_point(alpha = 0.4, size = 1.5) +
